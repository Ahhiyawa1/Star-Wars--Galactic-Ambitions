#Sandpeople on Tatooine
create_sandpeople_nation = {
	optimize_memory
	if = { limit = { NOT = { any_country = { is_country_type = adversary has_country_flag = sandpeople_nation } } }
		create_species = {
			name = "NAME_Sandpeople"
			class = TUS
			portrait = tusken_raider
			traits = {
				trait = trait_nomadic
				trait = trait_resilient
				ideal_planet_class = "pc_desert"
			}
			homeworld = root
		}
		create_country = {
			name = "NAME_Sandpeople_Nation"
			species = last_created_species
			type = adversary
			ethos = {
				ethic = "ethic_xenophobe"
			}
			auto_delete = no
			flag = {
				icon = { category = "zoological" file = "flag_zoological_10.dds" }
				background= { category = "backgrounds" file = "new_dawn.dds" }
				colors = { "red" "black" "null" "null" }
			}
			effect = {
				set_country_flag = sandpeople_nation
				establish_contact = { who = event_target:sandpeople_planet_owner location = root }
				establish_communications_no_message = event_target:sandpeople_planet_owner
				save_event_target_as = sandpeople_nation
				owner_species = { save_event_target_as = sandpeople_species }
			}
		}
	}
	else_if = { limit = { NOT = { exists = event_target:sandpeople_nation } }
		random_country = {
			limit = {
				is_country_type = adversary
				has_country_flag = sandpeople_nation
			}
			save_event_target_as = sandpeople_nation
		}
	}
}

# Starting shipyard script
create_starting_shipyard = {
	save_event_target_as = shipyard_site
	solar_system = {
		set_star_flag = sw_shipyard_built
		spawn_megastructure = {
			type = swp_$TYPE|orbital$_shipyard_$SIZE|0$
			coords_from = event_target:shipyard_site
			owner = $OWNER$
			graphical_culture = $OWNER$
		}
		every_system_megastructure = { limit = { is_megastructure_type = swp_$TYPE|orbital$_shipyard_$SIZE|0$ }
			set_name = { key = "NAME_megastructure_shipyard_start" variable_string = "\\[shipyard_site.GetName]" }
		}
	}
	set_planet_flag = has_megastructure
}
